        function init() {

            // Populate Class Dropdown

            Object.keys(schoolData).forEach(cls => {

                const opt = document.createElement('option');

                opt.value = cls; opt.innerText = cls;

                els.classSel.appendChild(opt);

            });


            // Create Mission Buttons

            Object.keys(wordDatabase).forEach(cat => {

                const btn = document.createElement('button');

                btn.className = 'btn';

                btn.innerText = cat;

                btn.onclick = () => startGame(cat);

            });


            window.addEventListener('keydown', e => state.keys[e.key] = true);

            window.addEventListener('keyup', e => state.keys[e.key] = false);

        }


        // --- LOGIN LOGIC ---

        function populateNameList() {

            const cls = els.classSel.value;

            els.dataList.innerHTML = '';

            els.nameIn.value = '';

            

            if(cls && schoolData[cls]) {

                els.nameIn.disabled = false;

                schoolData[cls].forEach(name => {

                    const opt = document.createElement('option');

                    opt.value = name;

                    els.dataList.appendChild(opt);

                });

            } else {

                els.nameIn.disabled = true;

            }

        }


        function handleLogin() {

            const cls = els.classSel.value;

            const name = els.nameIn.value;

            

            // Validate

            if(!cls || !name) { alert("Please select Class and Name!"); return; }

            

            // Check if name is roughly in the list (allows for minor typing case diffs)

            const validNames = schoolData[cls];

            const match = validNames.find(n => n.toUpperCase() === name.toUpperCase());


            if (!match) {

                if(!confirm("Name not found in list. Play as Guest with this name?")) return;

            }


            state.user.class = cls;

            state.user.name = match || name;

            document.getElementById('player-name-display').innerText = state.user.name.split(' ')[0]; // Show first name only

            document.getElementById('welcome-name').innerText = state.user.name;

            

            els.login.classList.add('hidden');

            els.start.classList.remove('hidden');

        }


        function startGame(cat) {

            state.targetCategory = cat;

            state.score = 0; 

            state.lives = 3; 

            state.bossActive = false; 

            state.entities = [];

            state.levelSpeed = 1;

            state.isPlaying = true;

            

            els.bossContainer.style.right = "-300px";

            els.start.classList.add('hidden');

            els.over.classList.add('hidden');

            

            document.getElementById('target-display').innerText = cat;

            document.getElementById('score-display').innerText = "0";

            document.getElementById('lives-display').innerText = "3";

        }


        // --- CORE LOOP ---

        function loop() {

            if(state.isPlaying) update();

            requestAnimationFrame(loop);

        }


        function update() {

            // Hero

            const speed = 6;

            if(state.keys.ArrowUp && state.heroPos.y > 0) state.heroPos.y -= speed;

            if(state.keys.ArrowDown && state.heroPos.y < window.innerHeight-60) state.heroPos.y += speed;

            if(state.keys.ArrowRight && state.heroPos.x < window.innerWidth-80) state.heroPos.x += speed;

            if(state.keys.ArrowLeft && state.heroPos.x > 0) state.heroPos.x -= speed;

            els.hero.style.top = state.heroPos.y + 'px'; 

            els.hero.style.left = state.heroPos.x + 'px';


            // Boss Logic

            if(state.bossActive) {

                state.bossY += 2 * state.bossDir;

                if(state.bossY > window.innerHeight - 200) state.bossDir = -1;

                if(state.bossY < 50) state.bossDir = 1;

                els.bossContainer.style.top = state.bossY + 'px';

            }


            // Spawn

            const spawnRate = state.bossActive ? 0.05 : 0.02;

            if(Math.random() < spawnRate) spawnWord();


            // Entities

            state.entities.forEach((ent, i) => {

                ent.x -= (4 + state.levelSpeed); 

                ent.el.style.left = ent.x + 'px';

                

                // Collision

                if(isColliding(els.hero, ent.el)) {

                    if(ent.cat === state.targetCategory) {

                        handleCorrectHit();

                    } else {

                        handleWrongHit();

                    }

                    ent.el.remove();

                    state.entities.splice(i, 1);

                } else if(ent.x < -200) {

                    ent.el.remove();

                    state.entities.splice(i, 1);

                }

            });

        }


        function spawnWord() {

            const isTarget = Math.random() > 0.4;

            const cats = Object.keys(wordDatabase);

            const cat = isTarget ? state.targetCategory : cats[Math.floor(Math.random()*cats.length)];

            const word = wordDatabase[cat][Math.floor(Math.random()*wordDatabase[cat].length)];

            

            const div = document.createElement('div');

            div.className = 'word-bubble';

            div.innerText = word;

            

            // Visuals

            if(cat === state.targetCategory) {

                div.style.backgroundColor = '#C8E6C9';

                div.style.border = '2px solid green';

            } else {

                div.style.backgroundColor = '#FFCDD2';

                div.style.border = '2px solid red';

            }


            // Position

            let startY = Math.random() * (window.innerHeight-50);

            let startX = window.innerWidth;


            if(state.bossActive) {

                startY = state.bossY + 100; // Shoot from boss

                startX = window.innerWidth - 200;

            }


            div.style.top = startY + 'px';

            div.style.left = startX + 'px';

            document.getElementById('game-container').appendChild(div);

            

            state.entities.push({ el: div, x: startX, cat: cat });

        }


        function isColliding(a, b) {

            const r1 = a.getBoundingClientRect();

            const r2 = b.getBoundingClientRect();

            return !(r1.right < r2.left || r1.left > r2.right || r1.bottom < r2.top || r1.top > r2.bottom);

        }


        // --- GAME EVENTS ---

        function handleCorrectHit() {

            state.score += 10;

            document.getElementById('score-display').innerText = state.score;


            if(state.bossActive) {

                state.bossHP--;

                els.bossHPFill.style.width = (state.bossHP/5)*100 + "%";

                if(state.bossHP <= 0) endBoss();

            } else {

                // Trigger Boss every 50 pts

                if(state.score > 0 && state.score % 50 === 0) startBoss();

            }

        }


        function handleWrongHit() {

            state.lives--;

            document.getElementById('lives-display').innerText = state.lives;

            els.hero.style.background = 'black';

            setTimeout(() => els.hero.style.background = 'linear-gradient(to right, #D32F2F, #B71C1C)', 100);

            

            if(state.lives <= 0) gameOver();

        }


        function startBoss() {

            state.bossActive = true;

            state.bossHP = 5;

            els.bossHPFill.style.width = "100%";

            els.bossContainer.style.right = "50px";

            // Flash Effect

            document.body.style.backgroundColor = "#333";

            setTimeout(() => document.body.style.backgroundColor = "", 200);

        }


        function endBoss() {

            state.bossActive = false;

            els.bossContainer.style.right = "-300px";

            state.levelSpeed += 0.5;

            if(state.lives < 3) { state.lives++; document.getElementById('lives-display').innerText = state.lives; }

        }


        function gameOver() {

            state.isPlaying = false;

            els.over.classList.remove('hidden');

            document.getElementById('final-score').innerText = state.score;

            saveDataToGoogleSheets();

        }


        // --- GOOGLE SHEETS SAVE ---

        function saveDataToGoogleSheets() {

            if(GOOGLE_SCRIPT_URL.includes("PASTE_YOUR")) {

                els.saveStatus.innerText = "Error: Google Sheet URL not set in code.";

                return;

            }


            const formData = new FormData();

            formData.append('name', state.user.name);

            formData.append('class', state.user.class);

            formData.append('score', state.score);


            fetch(GOOGLE_SCRIPT_URL, { method: 'POST', body: formData })

                .then(response => {

                    els.saveStatus.innerText = "✅ Score Saved to Dr. Saufi's Database!";

                    els.saveStatus.style.color = "#69F0AE";

                    els.saveStatus.classList.remove('blink');

                })

                .catch(error => {

                    els.saveStatus.innerText = "⚠️ Offline Mode (Saved locally)";

                });

        }


        init();

    </script>
